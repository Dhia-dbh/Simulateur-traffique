    def ajouter_vehicule(self, vehicule: "Vehicule") -> None:
        """Ajoute un véhicule à la route si celui-ci n'est pas déjà présent."""
        if vehicule not in self.vehicules_presents:
            self.vehicules_presents.append(vehicule)
